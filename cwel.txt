/*
UI FACTORIES

function checkbox(name){

const wrap=document.createElement("label");

wrap.innerHTML=`${name}<input type="checkbox">`;

const input=wrap.querySelector("input");

input.checked=MOD[name];

input.onchange=e=>MOD[name]=e.target.checked;

return wrap;

}


function slider(name,min,max){

const wrap=document.createElement("label");

const step = Number.isInteger(MOD[name]) ? 1 : 0.1;

wrap.innerHTML=`
${name}
<input type="range" min="${min}" max="${max}" step="${step}">
`;

const input=wrap.querySelector("input");

input.value=MOD[name];

input.oninput=e=>MOD[name]=parseFloat(e.target.value);

return wrap;

}


function dropdown(name,list){

const wrap=document.createElement("label");

const select=document.createElement("select");

list.forEach(v=>{

const opt=document.createElement("option");

opt.textContent=v;

select.appendChild(opt);

});

select.value=MOD[name];

select.onchange=e=>MOD[name]=e.target.value;

wrap.append(name,select);

return wrap;

}


function color(name,value){

const wrap=document.createElement("label");

const input=document.createElement("input");

input.type="color";

input.value=value;

input.onchange=e=>MOD[name]=e.target.value;

wrap.append(name,input);

return wrap;

}

btn.style.background="#120022";

}

navigator.clipboard.writeText(localStorage.getItem("token"));

}

if(label==="Change Token"){

localStorage.setItem("token",MOD.Token);

}

};

return btn;

}

/*
DRAG SYSTEM
*/

let drag=false;

let offsetX=0;

let offsetY=0;


menu.addEventListener("mousedown",e=>{

drag=true;

offsetX=e.clientX-menu.offsetLeft;

offsetY=e.clientY-menu.offsetTop;

});


document.addEventListener("mouseup",()=>drag=false);


document.addEventListener("mousemove",e=>{

if(!drag)return;

menu.style.left=(e.clientX-offsetX)+"px";

menu.style.top=(e.clientY-offsetY)+"px";

});
